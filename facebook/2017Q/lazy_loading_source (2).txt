#include <bits/stdc++.h>

using namespace std;

int a[1234567];

int main() {
  int tt;
  scanf("%d", &tt);
  for (int qq = 1; qq <= tt; qq++) {
    printf("Case #%d: ", qq);
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
      scanf("%d", a + i);
    }
    sort(a, a + n);
    reverse(a, a + n);
    int need = 0;
    int ans = n;
    for (int i = 0; i < n; i++) {
      need += 49 / a[i] + 1;
      if (need > n) {
        ans = i;
        break;
      }
    }
    printf("%d\n", ans);
  }
  return 0;
}
